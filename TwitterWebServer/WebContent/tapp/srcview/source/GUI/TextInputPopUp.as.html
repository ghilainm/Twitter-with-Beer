package GUI
{
    import flash.display.DisplayObject;
    import flash.events.MouseEvent;
    
    import mx.controls.Label;
    import mx.controls.Spacer;
    import mx.effects.Fade;
    import mx.events.EffectEvent;
    import mx.managers.PopUpManager;
    
    import spark.components.BorderContainer;
    import spark.components.Button;
    import spark.components.HGroup;
    import spark.components.TextInput;
    import spark.components.VGroup;

    public class TextInputPopUp
    {
        private var fade:Fade;
        private var fct:Function;
        private var nbc:BorderContainer;
        private var tI:TextInput;
        
        public function TextInputPopUp(msg:String, but1:String, but2:String, fct:Function, dParent:DisplayObject)
        {
            this.fct = fct;
            
            fade = new Fade();
            fade.alphaFrom = 1;
            fade.alphaTo = 0;
            fade.duration = 1000;
            
            nbc = new BorderContainer();
            nbc.minHeight = 0;
            nbc.minHeight = 0;
            nbc.setStyle("hideEffect",startFading);
            nbc.addEventListener(EffectEvent.EFFECT_END,closePopUp);
            nbc.setStyle("cornerRadius",20);
            nbc.setStyle("borderWeight",2);
            nbc.setStyle("backgroundColor",0x330570);
            nbc.setStyle("backgroundAlpha",0.2);
            nbc.setStyle("borderColor",0xAC3BD4);
            nbc.setStyle("borderAlpha",0.8);
            
            var vg:VGroup = new VGroup();
            
            var hG0:HGroup = new HGroup();
            var message:Label = new Label();
            message.setStyle("backgroundColor",0xB764D4);
            message.text = msg;
            var hG0s0:Spacer = new Spacer();
            hG0s0.width=3;
            var hG0s1:Spacer = new Spacer();
            hG0s1.width=3;
            hG0.addElement(hG0s0);
            hG0.addElement(message);
            hG0.addElement(hG0s1);
            
            var vsp0:Spacer = new Spacer();
            vsp0.height = 3;
            
            var hG1:HGroup = new HGroup();
            var hG1s0:Spacer = new Spacer();
            hG1s0.width=3;
            var hG1s1:Spacer = new Spacer();
            hG1s1.width=3;
            tI = new TextInput();
            hG1.addElement(hG1s0);
            hG1.addElement(tI);
            hG1.addElement(hG1s1);
            
            var vsp1:Spacer = new Spacer();
            vsp1.height = 3;
            
            var hG:HGroup = new HGroup();
            var mSpacer1:Spacer = new Spacer();
            mSpacer1.width = 7;
            var buttonOne:Button = new Button();
            buttonOne.label = but1;
            buttonOne.addEventListener(MouseEvent.CLICK, startFunction);
            var butSpacer:Spacer = new Spacer();
            butSpacer.width = 20;
            var buttonTwo:Button = new Button();
            buttonTwo.label = but2;
            var mSpacer2:Spacer = new Spacer();
            mSpacer2.width = 7;
            buttonTwo.addEventListener(MouseEvent.CLICK, closePopUp);
            hG.addElement(mSpacer1);
            hG.addElement(buttonOne);
            hG.addElement(butSpacer);
            hG.addElement(buttonTwo);
            hG.addElement(mSpacer2);
            
            var mVSpacer1:Spacer = new Spacer();
            mVSpacer1.height = 5;
            var mVSpacer2:Spacer = new Spacer();
            mVSpacer2.height = 5;
            vg.addElement(mVSpacer1);
            vg.addElement(hG0);
            vg.addElement(hG1);
            vg.addElement(hG);
            vg.addElement(mVSpacer2);
            
            nbc.addElement(vg);
            
            PopUpManager.addPopUp(nbc,dParent,true);
            PopUpManager.centerPopUp(nbc);
        }
        
        private function closePopUp(x:Object):void{
            PopUpManager.removePopUp(nbc);
        }
        
        private function startFading(x:Object):void{
            nbc.visible = false;
        }
        
        private function startFunction(x:MouseEvent):void{
            fct(tI.text);
            startFading(null);
        }
    }
}
