public function createFollowingField(following:String, lineName:String):IVisualElement{
    var nhb:HGroup = new HGroup();
    var sp2:Spacer = new Spacer();
    sp2.height = 1;
    sp2.width = 10;
    nhb.addElement(sp2);
    var l:Label = new Label();
    l.text = following;
    l.setStyle("fontSize",22);
    nhb.addElement(l);
    var sp:Spacer = new Spacer();
    sp.height = 1;
    sp.width = 10;
    nhb.addElement(sp);
    nhb.addElement(createDeleteButtonFollowing(following,nhb,lineName));
    return nhb;
}

public function createDeleteButtonFollowing(following:String, ive:IVisualElement,lineName:String):Image{
    var dB:Image = new Image();
    dB.load(deleteButtonImageURL);
    dB.alpha = 0.85;
    var clickHandler:Function = function():void{
        var deleteFollowing:Function = function ():void{
            FollowingsVGroupDisplayerID.removeElement(ive);
            removeTweetFromAuthorFromLine(following,lineName);
            RemoveUserRequest(DisplayFollowingsLineChooser.selectedItem.label, following);
        }
        popUpDeleteFollower(following,deleteFollowing);
    }
    var onFocusHandler:Function = function(event:MouseEvent):void{
        dB.alpha = 1;
    }
    var onFocusOutHandler:Function = function(event:MouseEvent):void{
        dB.alpha = 0.85;
    }
    dB.addEventListener("click",clickHandler);
    dB.addEventListener(MouseEvent.MOUSE_OVER,onFocusHandler);
    dB.addEventListener(MouseEvent.MOUSE_OUT,onFocusOutHandler);
    return dB;
}

public function removeTweetFromAuthorFromLine(following:String, lineName:String):void{
    if(tweetByLineByAuthor[following] == null)return ;
    if(tweetByLineByAuthor[following][lineName] == null)return;
    var tab:ArrayList = tweetByLineByAuthor[following][lineName];
    for(var index:int=0;index<tab.length;index++){
        var tweetStr:String = tab.getItemAt(index,0) as String;
        if(tweetTabHolder[tweetStr+lineName] == null)continue;
        if(linesDictionnary[lineName+userNameLogged] == null)continue;
        linesDictionnary[lineName+userNameLogged].removeElement(tweetTabHolder[tweetStr+lineName]);
        tweetTabHolder[tweetStr+lineName] = null;
    }
    tweetByLineByAuthor[following][lineName] = null;
}

public function popUpDeleteFollower(following:String,deleteFollowing:Function):void{
    var popUpDeleteFollowerHandler:Function = function(evt:CloseEvent):void{
        if(evt.detail == Alert.OK)
            deleteFollowing();
    }
    Alert.show("Do you want to delete "+following+" from this line?","Attention",  Alert.OK|Alert.CANCEL,null, popUpDeleteFollowerHandler);
}

public function SuppressLinePopUp(lineName:String):void{
    var popUpDeleteFollowerHandler:Function = function(evt:CloseEvent):void{
        if(evt.detail == Alert.OK)
            SuppressLineOk(lineName);
    }
    Alert.show("Do you want to delete the line "+lineName+"?","Alert",  Alert.OK|Alert.CANCEL,null, popUpDeleteFollowerHandler);
}
    
public function SuppressLineOk(lineName:String):void{
    SuppressLineDbRequest(lineName);
}
