package GUI
{
    import flash.display.DisplayObject;
    import flash.events.KeyboardEvent;
    import flash.events.MouseEvent;
    import flash.events.TimerEvent;
    import flash.ui.Keyboard;
    import flash.ui.Mouse;
    import flash.utils.Timer;
    
    import mx.controls.Alert;
    import mx.controls.Spacer;
    import mx.core.FlexGlobals;
    import mx.events.EffectEvent;
    import mx.events.FlexMouseEvent;
    import mx.managers.PopUpManager;
    
    import spark.components.BorderContainer;
    import spark.components.HGroup;
    import spark.components.Label;
    import spark.components.VGroup;
    import spark.effects.Fade;

    public class PopUpFadingText
    {
        
        private var nbc:BorderContainer;
        private var fade:spark.effects.Fade;
        public function PopUpFadingText(msg:String, parent:DisplayObject, heightParent:int = 0, widthParent:int = 0)
        {
            fade = new Fade();
            fade.alphaFrom = 1;
            fade.alphaTo = 0;
            
            nbc = new BorderContainer();
            nbc.setStyle("cornerRadius",10);
            nbc.setStyle("borderWeight",1);
            nbc.setStyle("backgroundColor",0x2F0986);
            nbc.setStyle("backgroundAlpha",0.3);
            nbc.setStyle("borderColor",0x600CAC);
            nbc.setStyle("color",0x8D41D6);
            nbc.setStyle("borderAlpha",0.8);
            
            var vg:VGroup = new VGroup();
            var sp1:Spacer = new Spacer();
            sp1.height = 1;
            var sp2:Spacer = new Spacer();
            sp2.height = 1;
            
            var hg:HGroup = new HGroup();
            var sp3:Spacer = new Spacer();
            sp3.width = 7;
            var sp4:Spacer = new Spacer();
            sp4.width = 7;
            var labelToDisplay:Label = new Label();
            labelToDisplay.text = msg;
            hg.addElement(sp3);
            hg.addElement(labelToDisplay);
            hg.addElement(sp4);
            
            vg.addElement(sp1);
            vg.addElement(hg);
            vg.addElement(sp2);
            
            nbc.addElement(vg);
            nbc.setStyle("hideEffect",fade);
            nbc.addEventListener(FlexMouseEvent.MOUSE_DOWN_OUTSIDE,startFading);
            nbc.addEventListener(EffectEvent.EFFECT_END,closePopUp);
            nbc.addEventListener(KeyboardEvent.KEY_DOWN,closePopUpKeyDownHandler);
            nbc.minHeight = 0;
            nbc.minWidth = 0;
            PopUpManager.addPopUp(nbc,parent,false);
            PopUpManager.centerPopUp(nbc);
            nbc.x = nbc.x + widthParent;
            
            //Set time before fading
            var timer:Timer = new Timer(1500,1);
            timer.addEventListener(TimerEvent.TIMER_COMPLETE,startFading);
            timer.start();
        }
        
        private function closePopUp(x:Object):void{
            PopUpManager.removePopUp(nbc);
        }
        
        private function startFading(x:Object):void{
            nbc.visible = false;
        }
        
        private function closePopUpKeyDownHandler(event:KeyboardEvent):void{
            if(event.keyCode == Keyboard.ENTER  ||  event.keyCode== Keyboard.SPACE || event.keyCode == Keyboard.BACKSPACE){
                startFading(null);
            }
        }
    }
}
