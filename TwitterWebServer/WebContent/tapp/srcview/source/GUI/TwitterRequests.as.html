protected function TwitterServiceResultHandler(event:ResultEvent):void{
    var xmlResponse:XML = new XML(event.result);
    var meth:String = xmlResponse.meth;
    var indexAND:int;
    var idToResend:String = xmlResponse.idToResend.toString();
    var userNameExtracted:String;
    var lineName:String;
    var listName:String;
    if(xmlResponse.failed == "true"){
        Alert.show(xmlResponse.error);
    }
    else if(meth==moreTweetsFromLine || meth==refreshTweetsFromLine){
        //idToresend = lineName+"&"+userName
        //if userName = userName
        indexAND = idToResend.search("&");
        lineName = idToResend.slice(0,indexAND);
        userNameExtracted = idToResend.slice(indexAND+1);
        var l:Vector.<IVisualElement> = new Vector.<IVisualElement>(); 
        for each(var tweet:XML in xmlResponse.descendants("TwitterApiImplementation.Tweet")){
            l.push(createTweet(tweet, lineName, false));
        }
        
        var nbrElem:int = (linesDictionnary[lineName+userNameExtracted]).numElements;
        while(l.length > 0){
            if(meth==moreTweetsFromLine)
                linesDictionnary[lineName+userNameExtracted].addElementAt(l.pop(),nbrElem);
            else 
                linesDictionnary[lineName+userNameExtracted].addElementAt(l.pop(),0);
            nbrElem++;
        }
    }
    else if(meth == refreshTweetsFromList || meth == moreTweetsFromList){
        //idToresend = listName
        listName = idToResend;
        var tweetVector:Vector.<IVisualElement> = new Vector.<IVisualElement>(); 
        for each(var tweet1:XML in xmlResponse.descendants("TwitterApiImplementation.Tweet")){
            tweetVector.push(createTweet(tweet1, listName, true));
        }
        while(tweetVector.length > 0){
            listesDictionnary[listName+userNameLogged].addElementAt(tweetVector.pop(),0);
        }
    }
    else if(meth==getLineNames){
        for each(var str:String in xmlResponse.descendants("string")){
            createLineGui(str);
        }
        LinesViewStack.selectedIndex = 0;
        LineChooser.selectedIndex = 0;
    }
    else if(meth == getListNames){
        var iL:ArrayList = new ArrayList();
        for each(var listName1:String in xmlResponse.descendants("string")){
            var xxx:Object = new Object();
            xxx.label = listName1;
            xxx.rollOverColor = 0x8D41D6;
            iL.addItem(xxx);
            listArray.addItem(xxx);
        }
        FavorisChooserList.initIMDDM(iL, TwitterSessionID);
    }
    else if(meth==logIn || meth==createUser ){
        TwitterSessionID = xmlResponse.twitterSessionID;
        userNameLogged = xmlResponse.idToResend;
        logInOk();
    }
    else if(meth==postTweet || meth==addUser || meth == removeUser || meth == addTweetToList){
    }
    else if(meth == createLine){
        createLineGui(xmlResponse.idToResend);
    }
    else if(meth == deleteLine){
        for(var i:int = 0 ;i < lineArray.length ; i++){
            if(lineArray.getItemAt(i,0).label == xmlResponse.idToResend){
                lineArray.removeItemAt(i);
                i = lineArray.length;
            }
        }
        FollowingsVGroupDisplayerID.removeAllElements();
        DisplayFollowingsLineChooser.selectedIndex = 0;
        LineChooser.selectedIndex = 0;
    }
    else if(meth == deleteList){
        for(var listIndex:int = 0 ;listIndex < lineArray.length ; listIndex++){
            if(listArray.getItemAt(listIndex,0).label == xmlResponse.idToResend){
                listArray.removeItemAt(listIndex);
                listIndex = listArray.length;
            }
        }
        ListeChooser.selectedIndex = 0;
    }
    else if(meth==allUsersFromLine){
        //idToresend = lineName+"&"+userName
        //if userName = userNameLogged
        idToResend = xmlResponse.idToResend.toString();
        indexAND = idToResend.search("&");
        userNameExtracted = idToResend.slice(indexAND+1);
        lineName = idToResend.slice(0,indexAND);
        if(userNameExtracted == userNameLogged){
            FollowingsVGroupDisplayerID.removeAllElements();
            for each(var following:String in xmlResponse.descendants("string")){
                FollowingsVGroupDisplayerID.addElement(createFollowingField(following, lineName));
            }
        }
        else {
            // FOR LATER
        }
    }
    else if(meth == disconnect){
        navigateToURL(new URLRequest(MAINURL),'_self');
    }
}


protected function MoreTweetsFromLine(lineName:String, userName:String):void{

    if(!validateInput(userName))return;
    var twitterServiceRequest:Object = new Object();
    twitterServiceRequest.appid = "TwitterAPI";
    twitterServiceRequest.meth = moreTweetsFromLine;
    twitterServiceRequest.nbrArgs = "2";
    twitterServiceRequest.idToResend = lineName+"&"+userName;
    twitterServiceRequest.arg0 = lineName;
    twitterServiceRequest.arg1 = userName;
    twitterServiceRequest.twitterSessionID = TwitterSessionID;
    TwitterService.request = twitterServiceRequest;
    TwitterService.send();
}

protected function validateInput(input:String):Boolean{
    var excludeChar:RegExp = /&/;
    if(input.search(excludeChar)==-1)
        return true;
    else{
        Alert.show("Please do not use any characters from "+excludeChar);
        return false;
    }
}
