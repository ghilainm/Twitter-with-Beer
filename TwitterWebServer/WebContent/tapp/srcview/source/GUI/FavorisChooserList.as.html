package GUI
{
    import flash.display.DisplayObject;
    import flash.events.MouseEvent;
    import flash.geom.Point;
    
    import mx.collections.ArrayList;
    import mx.collections.IList;
    import mx.controls.Alert;
    import mx.core.ClassFactory;
    import mx.core.FlexGlobals;
    import mx.core.IFactory;
    import mx.events.EffectEvent;
    import mx.events.FlexMouseEvent;
    import mx.events.ItemClickEvent;
    import mx.events.ListEvent;
    import mx.managers.PopUpManager;
    import mx.messaging.Channel;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;
    import mx.rpc.http.HTTPService;
    import mx.utils.ArrayUtil;
    
    import spark.components.List;
    import spark.components.supportClasses.Skin;
    import spark.effects.Fade;
    
    public class FavorisChooserList extends List
    {
        private static var httpRequestService:HTTPService = new HTTPService();
        private static var  addTweetToList:String =  'addTweetToList';
        private var tweetID:String;
        private static var dataProv:ArrayList;
        private var fade:Fade;
        private static var TwitterSessionID:String;
        
        private static function initHTTPService():void{
            httpRequestService.method="GET";
            httpRequestService.addEventListener(ResultEvent.RESULT,AddTweetToListRequestResultHandler);
            httpRequestService.addEventListener(FaultEvent.FAULT,AddTweetToListRequestFaultHandler);
            httpRequestService.showBusyCursor=true;
            httpRequestService.resultFormat="text";
            //httpRequestService.url = "https://87.65.179.139:443/TwitterWebServer/TwitterApiServlet";
        }
        
        public static function updateIP(ip:String):void{
            httpRequestService.url = ip+"/TwitterWebServer/TwitterApiServlet";
        }
        
        private static function initDataProv(l:IList):void{        
            dataProv = new ArrayList();
            for(var i:int;i<l.length;i++){
                dataProv.addItem(l.getItemAt(i,0));
            }
            
        }
        
        public static function initIMDDM(l:IList, tsi:String):void{
            TwitterSessionID = tsi;
            initHTTPService();
            initDataProv(l);
        }
        
        public function FavorisChooserList(tweetID:String, t:DisplayObject, xParent:int, yParent:int,widhtParent:int, heightParent:int)
        {
            fade = new Fade();
            fade.alphaFrom = 1;
            fade.alphaTo = 0;
            
            this.tweetID = tweetID;
            this.dataProvider = dataProv;
            addEventListener(FlexMouseEvent.MOUSE_DOWN_OUTSIDE,closePopUp);
            addEventListener(ItemClickEvent.ITEM_CLICK,itemClickedHandler);
            addEventListener(EffectEvent.EFFECT_STOP,closePopUp);
            this.setStyle("hideEffect",fade);
            
            PopUpManager.addPopUp(this,t,false);
            this.x = xParent+widhtParent;
            this.y = yParent;
            this.alpha = 0.9;
            this.itemRenderer = new ClassFactory(plusButtonItem);
            this.setStyle("borderVisible","true");
            this.setStyle("borderAlpha",0.5);
            this.setStyle("borderWeight",2);
            this.setStyle("contentBackgroundColor",0x2F0986);
            var colorArray:Array = [0x7A2EC8,0x8646C8];
            this.setStyle("alternatingItemColors",colorArray);
        }
        
        override public function stylesInitialized():void {
            super.stylesInitialized();
            this.setStyle("skinClass",Class(roundedListSkin));
        }
        
        private function startFading(x:Object):void{
            this.visible = false;
        }
        
        private function closePopUp(x:Object):void{
            PopUpManager.removePopUp(this);
        }
        
        public function itemClickedHandler(evt:ItemClickEvent):void{
            AddTweetToListRequest(tweetID,evt.item.label);
            new PopUpFadingText("This tweet has been added to "+evt.item.label,this,height, width);
            startFading(null);
        }
        
        private static function validateInput(input:String):Boolean{
            var excludeChar:RegExp = /&/;
            if(input.search(excludeChar)==-1)
                return true;
            else{
                Alert.show("Please do not use any characters from "+excludeChar);
                return false;
            }
        }
        
        private static function AddTweetToListRequest(tweetID:String, listName:String):Boolean{
            if(!validateInput(listName))return false;
            if(!validateInput(tweetID))return false;
            var twitterServiceRequest:Object = new Object();
            twitterServiceRequest.appid = "TwitterAPI";
            twitterServiceRequest.meth = addTweetToList;
            twitterServiceRequest.idToResend = "";
            twitterServiceRequest.twitterSessionID = TwitterSessionID;
            twitterServiceRequest.nbrArgs = "2";
            twitterServiceRequest.arg0 = listName;
            twitterServiceRequest.arg1 = tweetID;
            httpRequestService.request = twitterServiceRequest;
            httpRequestService.send();
            return true;
        }
        
        private static function AddTweetToListRequestResultHandler(event:ResultEvent):void{
            var xmlResponse:XML = new XML(event.result);
            var meth:String = xmlResponse.meth;
            if(xmlResponse.failed == "true"){
                Alert.show(xmlResponse.error);
            }
        }
        
        private static function AddTweetToListRequestFaultHandler(event:FaultEvent):void{
            Alert.show(event.fault.toString());    
        }
    }
}
