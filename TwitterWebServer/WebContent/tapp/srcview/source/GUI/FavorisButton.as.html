package GUI
{
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.geom.Point;
    
    import mx.controls.Alert;
    import mx.controls.Image;

    public class FavorisButton extends Image
    {
        private static var favorisButtonImage:String = "../images/favorisButtonIcon.png";
        private var tweetID:String;
        
        public function FavorisButton(tweetID:String)
        {
            this.tweetID = tweetID;
            this.load(favorisButtonImage);
            this.alpha = 0.85;
            this.tweetID = tweetID;
            this.addEventListener(MouseEvent.CLICK,imageClickHandler);
            this.addEventListener(MouseEvent.MOUSE_OVER,onFocusHandler);
            this.addEventListener(MouseEvent.MOUSE_OUT,onFocusOutHandler);
        }
        
        public function imageClickHandler(event:MouseEvent):void{
            var myGlobalPos:Point = localToGlobal(new Point(x,y));
            new FavorisChooserList(tweetID,this,myGlobalPos.x,myGlobalPos.y,width, height);
        }
        
        private function onFocusHandler(event:MouseEvent):void{
            this.alpha = 1;
        }
        
        private function onFocusOutHandler(event:MouseEvent):void{
            this.alpha = 0.85;
        }
    }
}
